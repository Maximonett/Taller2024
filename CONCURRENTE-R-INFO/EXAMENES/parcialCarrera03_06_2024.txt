  comenzar
    mientras HayFlorEnLaEsquina

  fin
  proceso juntarPapeles(ES p:numero)
  comenzar
    mientras HayPapelEnLaEsquina
      tomarPapel
      p:=p+1
  fin
  proceso juntarUnaFlor(ES f:numero;ES noHayMas:boolean)
  comenzar
    si HayFlorEnLaEsquina   
      tomarFlor
      f:=f+1
    sino
      noHayMas:=V
  fin

  robot corredor1
  variables
    id,f,p:numero
    noHayMas:boolean
  comenzar
    RecibirMensaje(id,fiscal)
    derecha
    repetir 4
      juntarFlores(f)
      mover
    noHayMas:=F  
    mientras (noHayMas=F)
      BloquearEsquina(6,6)
      Pos(6,6)
      juntarUnaFlor(f,noHayMas)
      Pos(5,7)
      LiberarEsquina(6,6)
    Pos(7,5)
    repetir 4
      juntarFlores(f)
      mover
    atras
    EnviarMensaje(id,fiscal)
    EnviarMensaje(f,fiscal)
    repetir 4
      juntarPapeles(p)
      mover
    noHayMas:=F
    mientras(noHayMas)
      BloquearEsquina(6,6)
      Pos(6,6)
      juntarUnPapel(p,noHayMas)
      Pos(7,5)
      LiberarEsquina(6,6)
    Pos(5,7)
    repetir 4
      juntarPapeles(p)
      mover
    EnviarMensaje(id,fiscal)
    EnviarMensaje(p,fiscal)
  fin

  robot fiscalizador
  variables
    id,f1,p1,f2,p2:numero
  comenzar
    EnviarMensaje(1,r1)
    EnviarMensaje(2,r2)
    repetir 2
      RecibirMensaje(id,*)
      si (id=1)
        RecibirMensaje(f1,r1)
      sino
        RecibirMensaje(f2,r2)
    si (f1>f2)
      Informar('ElRobotQueJuntoMasFloresFue',1)
    sino
      Informar('ElRobotQueJuntoMasFloresFue',2)
    repetir 2
      RecibirMensaje(id,*)

      sino

  fin

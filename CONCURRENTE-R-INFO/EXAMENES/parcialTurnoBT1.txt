programa parcialTurnoBT1
procesos
  proceso juntarFlores(ES f:numero)
  comenzar
    mientras HayFlorEnLaEsquina
      tomarFlor
      f:=f+1
  fin
areas
  area1:AreaP(1,1,100,100)
robots
  robot f1
  variables
    llegue:boolean
    f,id,X:numero 
  comenzar
    llegue:=V
    RecibirMensaje(id,fiscalizador)
    RecibirMensaje(X,fiscalizador)
    juntarFlores(f)      
    repetir 59
      si (PosCa=(X-1))
        Informar(1)
        BloquearEsquina(6,X)
        mover
        EnviarMensaje(llegue,fiscalizador)
        EnviarMensaje(id,fiscalizador)
        juntarFlores(f)
        mover
        LiberarEsquina(6,X)
        juntarFlores(f)
      sino
        mover
        juntarFlores(f)
    EnviarMensaje(id,fiscalizador)
    EnviarMensaje(f,fiscalizador)              
  fin
  robot f2
  variables
    X,id,f:numero
    llegue:boolean
  comenzar
    llegue:=V
    RecibirMensaje(id,fiscalizador)
    RecibirMensaje(X,fiscalizador)
    Pos(1,X)
    derecha
    repetir 59
      si (PosAv=6)
        Informar(2)
        BloquearEsquina(6,X)
        mover
        EnviarMensaje(llegue,fiscalizador)
        EnviarMensaje(id,fiscalizador)
        juntarFlores(f)
        mover
        LiberarEsquina(6,X)
        juntarFlores(f)
      sino
        mover
        juntarFlores(f)
    EnviarMensaje(id,fiscalizador)
    EnviarMensaje(f,fiscalizador)   
  fin
  robot fiscalizador
  variables
    id,f,f1,f2,X,ganador,perdedor:numero
    ok:boolean
  comenzar
    EnviarMensaje(1,r1)
    EnviarMensaje(2,r2)
    Random(X,2,50)
    EnviarMensaje(X,r1)
    EnviarMensaje(X,r2)
    RecibirMensaje(ok,*)
    RecibirMensaje(id,*)
    si (id=1)
      ganador:=id
    sino
      si (id=2)
        ganador:=id
    RecibirMensaje(ok,*)
    RecibirMensaje(id,*)
    si (id=1)
      perdedor:=id
    sino
      si (id=2)
        perdedor:=id
    repetir 2
      RecibirMensaje(id,*)
      si id=1
        RecibirMensaje(f,r1)
        f1:=f
      sino
        RecibirMensaje(f,r2)
        f2:=f
    si ganador=1
      Informar('ganador',ganador)
      Informar('cantidadDeFlores',f1)
    sino
      si ganador=2
        Informar('ganador',ganador)
        Informar('cantidadDeFlores',f2)          
  fin    
  
variables
  r1:f1
  r2:f2
  fiscalizador:fiscalizador
comenzar
  AsignarArea(r1,area1)
  AsignarArea(r2,area1)
  AsignarArea(fiscalizador,area1)
  Iniciar(r1,6,1)
  Iniciar(r2,1,2)
  Iniciar(fiscalizador,100,100)
fin

programa eje3

    mientras HayFlorEnLaEsquina & f<10
      tomarFlor
      f:=f+1
  fin
  proceso depositarFlores
  comenzar
    mientras  HayFlorEnLaBolsa
      depositarFlor
  fin

robots

  variables
    ai,ci,f,esquina:numero
  comenzar
    derecha
    esquina:=0
    f:=0
    mientras (PosAv<100)
      mientras (PosAv<100) & (f<10) & (esquina<5)
        juntarFlores(f)
        mover
        esquina:=esquina+1
      ai:=PosAv
      ci:=PosCa

      si (esquinas=5)
        esquina:=0
      sino
        si (f=10)
          f:=0
      Pos(ai + 1,ci)

    EnviarMensaje(F,cosechador)    
  fin
  robot cosechador

    ahora,ok:boolean

  comenzar
    ok:=V

      RecibirMensaje(ahora,*)

      Pos(1,5)
